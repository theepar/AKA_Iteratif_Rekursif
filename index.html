// Algoritma Iteratif
public static String caraIteratif(int n) {
    if (n == 0) return "0";
    String biner = "";
    while (n > 0) {
        biner = (n % 2) + biner; 
        n = n / 2;
    }
    return biner;
}

// Algoritma Rekursif
public static String caraRekursif(int n) {
    if (n == 0) return "";
    return caraRekursif(n / 2) + (n % 2);
}
                

// Algoritma Iteratif (JS)
function caraIteratif(n) {
    if (n === 0) return "0";
    let biner = "";
    while (n > 0) {
        biner = (n % 2) + biner;
        n = Math.floor(n / 2);
    }
    return biner;
}

// Algoritma Rekursif (JS)
function caraRekursif(n) {
    if (n === 0) return "";
    return caraRekursif(Math.floor(n / 2)) + (n % 2);
}
